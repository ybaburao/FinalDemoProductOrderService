package com.cts.mc.product.model;

import java.io.Serializable;

import javax.persistence.Column;
import javax.persistence.Entity;
import javax.persistence.GeneratedValue;
import javax.persistence.GenerationType;
import javax.persistence.Id;

@Entity
public class ProductEntity implements Serializable {
	/**
	 * 
	 */
	private static final long serialVersionUID = 1L;
	@Id
	@GeneratedValue(strategy = GenerationType.AUTO)
	private Long entityProductId;
	@Column(name = "productName")
	private String entityProductName;
	@Column(name = "stock")
	private int entityStock;
	@Column(name = "price")
	private Double entityPrice;
	@Column(name = "offer")
	private Double entityOffer;
	
	
	public ProductEntity() {
		super();
	}
	public ProductEntity(Long productId ,String productName, int stock, Double price, Double offer) {
		super();
		this.entityProductId=productId;
		this.entityProductName = productName;
		this.entityStock = stock;
		this.entityPrice = price;
		this.entityPrice = offer;
	}
	public Long getEntityProductId() {
		return entityProductId;
	}
	public void setEntityProductId(Long entityProductId) {
		this.entityProductId = entityProductId;
	}
	public String getEntityProductName() {
		return entityProductName;
	}
	public void setEntityProductName(String entityProductName) {
		this.entityProductName = entityProductName;
	}
	public int getEntityStock() {
		return entityStock;
	}
	public void setEntityStock(int entityStock) {
		this.entityStock = entityStock;
	}
	public Double getEntityPrice() {
		return entityPrice;
	}
	public void setEntityPrice(Double entityPrice) {
		this.entityPrice = entityPrice;
	}
	public Double getEntityOffer() {
		return entityOffer;
	}
	public void setEntityOffer(Double entityOffer) {
		this.entityOffer = entityOffer;
	}
	
	

}
